// Coin change - Brute Force Approach
#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n;
    cin>>n;
    vector<int> v(n);
    for(int i=0;i<n;i++)
    {
        cin>>v[i];
    }
    int amount;
    cin>>amount;
    int i=0;
    vector<int> res;
    sort(v.begin(),v.end(),greater<int>());
    while(amount>0 && i<n)
    {
        if(amount<v[i])
        {
            i++;
        }
        else
        {
            amount = amount - v[i];
            res.push_back(v[i]);
        }
    }
    cout<<res.size()<<endl;
    return 0;
}
